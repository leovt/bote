%ignore /[ \n\t]/

COMMENT: /\<[\w\s]+\>/
%ignore COMMENT


// the following rules will be defined by the program reading this grammar.
// these are just sample rules
// keyword: /flying|crush|readiness|cannot_block|cannot_attack/

color: /red|green|blue|yellow|white/
type: /player|permanent|creature|enchantment|source|instant|sorcery/
subtype: /goblin|dragon|elemental|minotaur|human|elf/
step: /end/
energy: /(\{([0-9]+|[RGBWY])\})+/
number: /[0-9]+/
signed_number: /[\+\-][0-9]+/
variable: "X"
signed_variable: /[\+\-]X/
_numeric: number | variable
_signed_numeric: signed_number | signed_variable

start: effects

effects: _effect (";" _effect)* [";"]

_effect: _immediate_effect
       | triggered_effect

triggered_effect: "when" trigger ":" _immediate_effect

trigger: _object "gets" "tapped" -> tap_trigger
       | _object "gets" "damage" -> damage_trigger
       | _object "gets" "blocked" "by" _object -> blocked_by_trigger
       | step "step" "begins" -> step_begin_trigger
       | "your" "turn" "begins" -> turn_begin_trigger
       | this "enters" "the" "battlefield" -> enters_battlefield_trigger
       | _object ("cast"|"casts") _object -> cast_trigger

_immediate_effect: damage_effect
       | add_energy_effect
       | destruction_effect
       | create_token_effect
       | continuous_effect
       | put_counter_effect
       | untap_effect
       | pay_or_destroy_effect

damage_effect: _object "gets" _numeric "damage"
destruction_effect: "destroy" _object
add_energy_effect: "add" energy "to" _object "energy pool"
create_token_effect: "create" _numeric "(" number ")" "token"
continuous_effect: _continuous_modifier [until_end_of_turn]
put_counter_effect: "put" "a" modifier "counter" "on" _object
untap_effect: "untap" _object
pay_or_destroy_effect: "pay" energy "or" destruction_effect

_continuous_modifier: _object "has" modifier_list
                    | prevent_battle_damage

prevent_battle_damage: "prevent" "all" "battle" "damage"

until_end_of_turn: "until end of turn"

modifier_list: modifier ("and" modifier)*

modifier: _signed_numeric "/" _signed_numeric -> delta_stat
        | _numeric "/" _numeric -> set_stat
        | keyword -> add_keyword
        | "not" keyword -> remove_keyword
        | "controlled by" _object -> change_controller

_object: all | this | effect_controller | _chosen | controller_of | enchanted | any

all: "all" selector
any: "any" selector [chosen_label]
this: "this"
effect_controller: "you" | "your"
_chosen: chosen_spec | chosen_ref

chosen_spec: "chosen" selector [chosen_label]
chosen_ref: _chosen_ref_label
chosen_label: _chosen_ref_label
 _chosen_ref_label: "[" /[\w]+/ "]"

enchanted: "enchanted" selector [chosen_label]

selector: selector_term+

selector_term: "." _spec -> positive_selector
             | "!" _spec -> negative_selector

_spec: type_spec | color_spec | other_spec | subtype_spec | keyword_spec

color_spec: "color" "(" color ("," color)* ")"
          | color

type_spec: "type" "(" type ("," type)* ")"
          | type

subtype_spec: "subtype" "(" subtype ("," subtype)* ")"
          | subtype

keyword_spec: keyword

other_spec: "other"

controller_of: "controller of" _object
